package editor
 
Buffer_View :: struct {
    buf: buffer.Buffer,
    cursor: cursor.Selection,
    scroll: int,
}
 
editor_open :: proc(state: ^Editor_State, path: string) {
    buff, ok := buffer.buffer_load_file(path)
    if !ok do return
    append(&state.views, Buffer_View{buf = buff})
}
